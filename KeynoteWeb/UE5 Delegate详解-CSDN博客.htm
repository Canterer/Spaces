C++语言将函数抽象为由返回值、参数类型组成的函数指针类型。可以通过操作指针来访问该函数，而无需关注指针里定义的具体实现。由此将定义函数指针逻辑的部分，与执行函数指针逻辑的部分彻底拆分开，从而实现函数的跨模块调用。

UE中，Delegate一般称为代理，只触发一个函数指针执行的行为称为单播，触发多个的称为多播。如果需要蓝图的支持，则称为动态代理。

代理是默认非异步的，即触发代理时，所有的函数指针会按照添加的顺序依次执行。若想实现异步的代理，我们需要额外使用UE中的异步执行功能，在异步代码块中触发代理。

UE的4类代理：静态单播、静态多播、动态单播、动态多播。没有一个共通的基类。多播代理的内部会存储一个对应的单播代理的TArray，而不是函数指针的TArray。而动态单播与静态单播则时完全不相关的两部分。

对于静态代理，存在基类FDelegateBase与FMulticaseDelegateBase(UE5废除)。通过TBaseDelegate与TBaseMulticastDelegate(UE5该类为静态多播基类)继承自上述两个类。

    {
        typedef TArray< TScriptDelegate<TWeakPtr>, TInlineAllocator< 4 > > FInlineInvocationList;
        FInlineInvocationList InvocationListCopy = FInlineInvocationList(InvocationList);
        for( typename FInlineInvocationList::TConstIterator FunctionIt( InvocationListCopy ); FunctionIt; ++FunctionIt )
        {
            if( FunctionIt->IsBound() )
                FunctionIt->template ProcessDelegate<UObjectTemplate>(Parameters);
            else if( FunctionIt->IsCompactable() )

UE的静态代理是以DelegateInstance为媒介实现的。实现一个泛用性很强的代理，需要克服的困难，一是对不同类型函数的兼容，二是对不定数量的参数数量的兼容。UE使用DelegateInstance的模板推导克服了这两个困难。IDelegateInstance是所有相关类的基类。

IBaseDelegateInstanceCommon(UE5废弃采用TCommonDelegateInstanceState)是一个以函数类型为模板参数的类，其接受函数模板的偏特化形式继承自IDelegateInstance, 并在此基础上再偏特化出来一个无返回值的形式。在函数形式的偏特化中，使用不定数量的模板参数，来接收不懂数量的函数输入参数。

IBaseDelegateInstance也是一个以函数类型为模板参数的类，继承自IBaseDelegateInstanceCommon<FuncType>(UE5废弃采用TCommonDelegateInstanceState), 使得自身的两种类型的偏特化能够直接继承自对应的基类偏特化形式。

在此基础上，分别定义七种不同类型的子类实现，每种类型都会额外偏特化出一个无返回值类型，来实现上述三层抽象定义的所有接口。

TBaseUFunctionDelegateInstance    接收UFunction
TBaseSPMethodDelegateInstance    接收共享引用类型的类成员函数
TBaseRawMethodDelegateInstance    接收普通类型的类成员函数
TBaseUObjectMethodDelegateInstance 接收UObject类型的类成员函数
TBaseStaticDelegateInstance    接收普通C++函数或类的静态函数
TBaseFunctionDelegateInstance    接收Lambda函数
TWeakBaseFunctionDelegateInstance    接收类成员Lambda函数
以上是UE能够支持的所有函数类型的代理。他们自身实现了每一种函数的执行方式。每个Instance中有一个FDelegateHandle类型的变量，以Guid记录Delegate的唯一标志。每一个实现中都有一个不同的Create函数，会根据函数类型、参数数量的不同来进行Instance的内存分配与创建。

template< typename T >
struct TPlacementNewer
{
    TPlacementNewer() : bConstructed(false) {}
    ~TPlacementNewer()
    {
        if(bConstructed)
        {
            //We need a typedef here because VC won't compile the destructor call below if T itself has a member called T
            typedef T TPlacementNewerTTypeTypedef;
            ((TPlacementNewerTTypeTypedef*)&Bytes)->TPlacementNewerTTypeTypedef::~TPlacementNewerTTypeTypedef();
        }

    }

    template <typename... ArgTypes>
    T* operator()(ArgTypes&&... Args)
    {
        check(!bConstructed);
        T* Result = new (&Bytes) T(Forward<ArgTypes>(Args)...);
        bConstructed = true;
        return Result;
    }

    T* operator->()
    {
        check(bConstructed);
        return (T*)&Bytes;
    }

private:
    TTypeCompatibleBytes<T> Bytes;
    bool bConstructed;
}

// Declare user's dynamic delegate, with wrapper proxy method for executing the delegate
// ExecFunction  DelegateName##_DelegateWrapper
// FuncParamList    FUNC_CONCAT(Param1Type InParam1, Param2Type InParam2)
// FuncParamPassThru     FUNC_CONCAT( *this, InParam1, InParam2 )
// ...    对应的就是 函数类型   ReturnValueType, Param1Type, Param2Type
// FUNC_DECLARE_DYNAMIC_DELEGATE 用于无返回值的 所以Execute的返回类型为void
#define FUNC_DECLARE_DYNAMIC_DELEGATE( TWeakPtr, DynamicDelegateClassName, ExecFunction, FuncParamList, FuncParamPassThru, ...) \
    class DynamicDelegateClassName : public TBaseDynamicDelegate<TWeakPtr, __VA_ARGS__> \
    {    \
    public: \
        inline void Execute( FuncParamList ) const \
        { \
            checkSlow( IsBound() ); \
            ExecFunction( FuncParamPassThru );
        } \
    };

class TDelegate

template<typename InRetValType, typename... ParamTypes, typename UserPolicy>
class TDelegate<InRetValType(ParamTypes...), UserPolicy> : public TDelegateBase<UserPolicy>
{
    using Super = TDelegateBase<UserPolicy>;
    using FuncType = InRetValType (ParamTypes...):
    using DelegateInstanceInterfaceType = IBaseDelegateInstance<FuncType, UserPolicy>
    
    using DelegateType = typename UserPolicy::FDelegateExtras;
    using DelegateInstanceType = typename UserPolicy::FDelegateInstanceExtras;
    using MulticastDelegateType = typename UserPolicy::FMulticastDelegateExtras;

public:
    //Make sure FDelegateBase's public funciton are publicly exposed through the TDelegate API
    using Super::Unbind;
private:
    using Super::GetDelegateInstanceProtected;
public:
    typedef InRetValType RetValType;
    typedef InRetValType TFuncType(ParamTypes...);

    // helper typedefs for getting a member function pointer  type for the delegate with a given payload
    template<typename UserClass, typename... VarTypes>
    using TMethodPtr = typename TMemFunPtrType<false, UserClass, RetValType(ParamTypes..., VarTypes...)>::Type;
    template<typename UserClass, typename... VarTypes>
    using TConstMethodPtr = typename TMemFunPtrType<true, UserClass, RetValType(ParamTypes...., VarTypes...)>::Type;

    //Declare the user's C++ pointer-based delegate instance types
    template<typename UserClass>
    struct TRawMethodDelegate : TBaseRawMethodDelegateInstance<false, UserClass, FuncType, UserPolicy>
    {
        typedef TBaseRawMethodDelegateInstance<false, UserClass, FuncType, UserPolicy> Super;
        TRawMethodDelegate(UserClass* InUserObject, typename Super::FMethodPtr InMethodPtr)
            : Super(InUserObject, InMethodPtr) {}
    }
    template<typename UserClass>
    struct TRawMethodDelegate_Const : TBaseRawMethodDelegateInstance<false, UserClass, FuncType, UserPolicy>
    {
        typedef TBaseRawMethodDelegateInstance<true, UserClass, FuncType, UserPolicy> Super;
        TRawMethodDelegate_Const(UserClass* InUserObject, typename Super::FMethodPtr InMethodPtr)
            : Super(InUserObject, InMethodPtr) {}
    }
    // one var
    template<typename UserClass, typename Var1Type>
    struct TRawMethodDelegate_OneVar : TBaseRawMethodDelegateInstance<false, UserClass, FuncType, UserPolicy, Var1Type>
    {
        typedef TBaseRawMethodDelegateInstance<false, UserClass, FuncType, UserPolicy, Var1Type> Super;
        TRawMethodDelegate_OneVar(UserClass* InUserObject, typename Super::FMethodPtr InMethodPtr, Var1Type Var1)
            : Super(InUserObject, InMethodPtr, Var1) {}
    }

    template<typename... VarTypes>
    struct FStaticDelegate 
        : TBaseStaticDelegateInstance<FuncType, UserPolicy, VarTypes...>
    {
        typedef TBaseStaticDelegateInstance<FuncType, UserPolicy, VarTypes...> Super;
        FStaticDelegate(typename Super::FFuncPtr InFuncPtr, VarTypes... Vars)
            : Super(InFuncPtr, Var...) {}
    }
    
public:
    template<typename... VarTypes>
    UE_NODISXARD inline static TDelegate<RetValType(ParamsTypes...), UserPolicy> CreateStatic(
    typename TIdentity<RetValType(*)(ParamTypes...， VarTypes...)>::Type InFunc,
    VarTypes... Vars)
    {
        TDelegate<RetValType(ParamTypes...), UserPolicy> Result;
        TBaseStaticDelegateInstance<FuncType, UserPolicy, VarTypes...>::Create(Result, InFunc, Vars...);
        return Result;
    }

    template<typename... VarTypes>
    inline void BindStatic(typename TBaseStaticDelegateInstance<FuncType, UserPolicy, VarTypes...>::FFuncPtr InFunc, VarTypes... Vars)
    { *this = CreateStatic(InFunc, Vars...); }
}
